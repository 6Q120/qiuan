        // 检查localStorage使用情况的函数
        function checkLocalStorageUsage() {
            try {
                let totalSize = 0;
                for (let key in localStorage) {
                    if (localStorage.hasOwnProperty(key)) {
                        totalSize += localStorage[key].length;
                    }
                }
                
                // 将字节转换为MB
                const totalMB = (totalSize / 1024 / 1024).toFixed(2);
                return {
                    totalSize: totalSize,
                    totalMB: parseFloat(totalMB),
                    percentage: Math.round((totalSize / 5242880) * 100) // 5MB = 5,242,880 bytes
                };
            } catch (error) {
                console.error('检查localStorage使用情况时出错:', error);
                return { totalSize: 0, totalMB: 0, percentage: 0 };
            }
        }
        
        // 清理旧消息的函数
        function cleanupOldMessages(maxMessages = 1000) {
            try {
                if (Array.isArray(messages) && messages.length > maxMessages) {
                    // 只保留最新的maxMessages条消息
                    const removedCount = messages.length - maxMessages;
                    messages = messages.slice(removedCount);
                    console.log(`已清理 ${removedCount} 条旧消息，当前保留 ${messages.length} 条`);
                    return true;
                }
                return false;
            } catch (error) {
                console.error('清理旧消息时出错:', error);
                return false;
            }
        }
        
        // 独立保存单个数据项的函数
        function saveSingleItem(key, data) {
            try {
                // 先验证数据结构，确保可以序列化
                if (data === undefined || data === null) {
                    console.warn(`尝试保存 ${key}，但数据为 ${data}`);
                    return false;
                }
                
                const serializedData = JSON.stringify(data);
                localStorage.setItem(key, serializedData);
                return true;
            } catch (error) {
                console.error(`保存 ${key} 到本地存储时出错:`, error);
                
                // 如果是因为存储空间不足导致的错误
                if (error.name === 'QuotaExceededError') {
                    console.log(`本地存储空间不足，尝试清理 ${key} 相关数据...`);
                    
                    // 根据不同的数据类型采取不同的清理策略
                    if (key === 'Q_messages') {
                        // 聊天记录过多，清理到更小的数量
                        if (cleanupOldMessages(300)) {
                            // 清理后重新尝试保存
                            console.log('清理旧消息后，重新尝试保存聊天记录...');
                            return saveSingleItem(key, data);
                        }
                    } else if (key === 'Q_localImages') {
                        // 表情包过多，清理旧的表情包
                        if (localImages.length > 50) {
                            const removedCount = localImages.length - 50;
                            localImages = localImages.slice(removedCount);
                            console.log(`已清理 ${removedCount} 个旧表情包，当前保留 ${localImages.length} 个`);
                            return saveSingleItem(key, localImages);
                        }
                    } else {
                        // 其他数据类型，尝试清理聊天记录
                        if (cleanupOldMessages(500)) {
                            // 清理后重新尝试保存
                            console.log('清理旧消息后，重新尝试保存数据...');
                            return saveSingleItem(key, data);
                        }
                    }
                    
                    // 所有清理策略都失败
                    console.error('无法清理足够的存储空间以保存数据');
                    return false;
                } else if (error.name === 'TypeError') {
                    // 数据无法序列化
                    console.error(`无法序列化 ${key} 的数据:`, data);
                    return false;
                }
                
                return false;
            }
        }

                if (saveToLocalStorageTimer) {
                    clearTimeout(saveToLocalStorageTimer);
                }
                
                // 使用独立保存函数确保所有数据都被保存
                try {
                    // 验证关键数据结构
                    if (!Array.isArray(messages)) messages = [];
                    if (!phrases || typeof phrases !== 'object') phrases = { text: [] };
                    if (!Array.isArray(localImages)) localImages = [];
                    
                    saveSingleItem('Q_phrases', phrases);
                    saveSingleItem('Q_localImages', localImages);
                    saveSingleItem('Q_messages', messages);
                    
                    // 保存设置
                    const settings = {
                        darkMode: domCache.darkModeToggle?.checked || false,
                        fontSize: domCache.fontSize?.value || 16,
                        primaryColor: getComputedStyle(document.documentElement).getPropertyValue('--primary-color').trim(),
                        replyDelay: domCache.replyDelay?.value || 0,
                        bgImage: (domCache.chatMessages?.style.backgroundImage || domCache.body?.style.backgroundImage).replace(/url\(['"]?(.*?)['"]?\)/i, '$1') || '',
                        fullscreenBg: domCache.fullscreenBgToggle?.checked || false,
                        bubbleEffect: domCache.bubbleEffect?.value || 100,
                        buttonOpacity: domCache.buttonOpacity?.value || 0.9,
                        buttonBlur: domCache.buttonBlur?.value || 5,
                        myName: domCache.myNameInput?.value || '',
                        contactName: domCache.contactNameInput?.value || '',
                        myAvatar: domCache.myAvatar?.src || `https://ui-avatars.com/api/?name=Me&background=${encodeURIComponent('#9dd6ff')}&color=fff&size=100`,
                        contactAvatar: domCache.contactAvatar?.src || `https://ui-avatars.com/api/?name=Q&background=${encodeURIComponent('#1e88e5')}&color=fff&size=100`
                    };
                    saveSingleItem('Q_settings', settings);
                    
                    // 在移动设备上，确保数据被持久化
                    if (navigator.storage && navigator.storage.persist) {
                        navigator.storage.persist().catch(err => {
                            console.warn('无法请求持久化存储:', err);
                        });
                    }
                } catch (error) {
                    console.error('页面隐藏时保存数据失败:', error);
                } finally {
                    saveToLocalStorageTimer = null;
                }
            }
        });

        // 计算消息统计数据
        function calculateStats() {
            try {
                // 获取聊天记录
                const messages = JSON.parse(localStorage.getItem('Q_messages') || '[]');
                
                // 确保messages是数组
                const validMessages = Array.isArray(messages) ? messages : [];
                
                // 计算总回复数（只统计对方的消息）
                // 支持多种发送者标识：'them'表示对方，同时确保过滤掉'me'以外的所有消息
                const theirMessages = validMessages.filter(msg => {
                    return msg && msg.sender !== 'me' && msg.sender !== undefined;
                });
                domCache.totalReplies.textContent = theirMessages.length;
                
                // 计算词库容量
                const savedPhrases = JSON.parse(localStorage.getItem('Q_phrases') || '{}');
                // 正确获取词库文本数量
                const phraseCount = savedPhrases && savedPhrases.text ? savedPhrases.text.length : 0;
                domCache.lexiconSize.textContent = phraseCount;
                
                // 计算初次相遇和最近联络时间（只统计对方的消息）
                if (theirMessages.length > 0) {
                    // 按时间排序
                    const sortedTheirMessages = [...theirMessages].sort((a, b) => {
                        try {
                            const timeA = new Date(a.time || 0).getTime();
                            const timeB = new Date(b.time || 0).getTime();
                            return timeA - timeB;
                        } catch (e) {
                            return 0;
                        }
                    });
                    
                    // 格式化日期
                    const formatDate = (dateString) => {
                        try {
                            const date = new Date(dateString);
                            return `${date.getFullYear()}-${(date.getMonth() + 1).toString().padStart(2, '0')}-${date.getDate().toString().padStart(2, '0')}`;
                        } catch (e) {
                            return '未知日期';
                        }
                    };
                    
                    // 设置初次相遇和最近联络
                    domCache.firstEncounter.textContent = formatDate(sortedTheirMessages[0].time);
                    domCache.recentContact.textContent = formatDate(sortedTheirMessages[sortedTheirMessages.length - 1].time);
                    
                    // 计算高频回复（只统计对方的消息）
                    const replyCounts = {};
                    theirMessages.forEach(msg => {
                        // 支持content和text字段，兼容旧数据结构
                        const content = msg.content || msg.text || '';
                        if (content.trim()) {
                            replyCounts[content] = (replyCounts[content] || 0) + 1;
                        }
                    });
                    
                    // 排序并获取前6个
                    const sortedReplies = Object.entries(replyCounts)
                        .sort((a, b) => b[1] - a[1])
                        .slice(0, 6);
                    
                    // 渲染高频回复
                    const topRepliesContainer = domCache.topReplies;
                    topRepliesContainer.innerHTML = '';
                    
                    sortedReplies.forEach(([content, count], index) => {
                        const replyItem = document.createElement('div');
                        replyItem.className = 'reply-item';
                        replyItem.innerHTML = `
                            <span class="reply-rank">${index + 1}.</span>
                            <span class="reply-content">${content}</span>
                            <span class="reply-count">${count}</span>
                        `;
                        topRepliesContainer.appendChild(replyItem);
                    });
                } else {
                    // 当没有对方消息时，清空统计数据
                    domCache.firstEncounter.textContent = '暂无数据';
                    domCache.recentContact.textContent = '暂无数据';
                    domCache.topReplies.innerHTML = '<div class="reply-item">暂无高频回复</div>';
                }
            } catch (error) {
                console.error('计算消息统计数据时出错:', error);
                // 出错时显示友好提示
                domCache.totalReplies.textContent = '0';
                domCache.lexiconSize.textContent = '0';
                domCache.firstEncounter.textContent = '统计出错';
                domCache.recentContact.textContent = '统计出错';
                domCache.topReplies.innerHTML = '<div class="reply-item">统计出错</div>';
            }
        }

                    isLongPress = false;
                });

                    currentEmojiItem = null;
                    isLongPress = false;
                });

                    isLongPress = false;
                });

                    clearTimeout(longPressTimer);
                    currentEmojiItem = null;
                    isLongPress = false;

                });

        // 添加消息到聊天
        function addMessageToChat(content, sender, time = null, type = 'text') {
            // 创建日期对象
            const now = new Date();
            let shortTime, fullTime;
            
            if (!time) {
                // 格式化短时间 (HH:MM)
                shortTime = `${now.getHours().toString().padStart(2, '0')}:${now.getMinutes().toString().padStart(2, '0')}`;
                // 格式化完整时间 (MM月DD日 周X HH:MM)
                const days = ['周日', '周一', '周二', '周三', '周四', '周五', '周六'];
                fullTime = `${(now.getMonth() + 1).toString().padStart(2, '0')}月${now.getDate().toString().padStart(2, '0')}日 ${days[now.getDay()]} ${shortTime}`;
            } else {
                // 如果提供了时间，使用它作为短时间，生成一个示例完整时间
                shortTime = time;
                const days = ['周日', '周一', '周二', '周三', '周四', '周五', '周六'];
                fullTime = `${(now.getMonth() + 1).toString().padStart(2, '0')}月${now.getDate().toString().padStart(2, '0')}日 ${days[now.getDay()]} ${shortTime}`;
            }
            
            // 获取当前时间
            const currentTime = new Date();
            
            // 计算时间差
            const fifteenMinutes = 15 * 60 * 1000; // 15分钟（毫秒）
            let showTimeLabel = domCache.chatMessages.children.length === 0;
            
            // 如果不是第一条消息，检查时间差
            if (!showTimeLabel && lastMessageTime) {
                const timeDiff = currentTime - lastMessageTime;
                showTimeLabel = timeDiff > fifteenMinutes;
            }
            
            // 使用requestAnimationFrame优化DOM操作
            requestAnimationFrame(() => {
                // 创建文档片段，减少DOM操作次数
                const fragment = document.createDocumentFragment();
                
                if (showTimeLabel) {
                    // 创建时间标签
                    const timeLabelDiv = document.createElement('div');
                    timeLabelDiv.className = 'time-label';
                    timeLabelDiv.textContent = shortTime;
                    timeLabelDiv.dataset.fullTime = fullTime;
                    
                    // 添加点击事件监听
                    timeLabelDiv.addEventListener('click', function() {
                        // 创建或切换完整时间显示
                        const existingFullTime = this.querySelector('.full-time');
                        if (existingFullTime) {
                            // 如果已经显示完整时间，移除它
                            existingFullTime.remove();
                        } else {
                            // 显示完整时间
                            const fullTimeDiv = document.createElement('div');
                            fullTimeDiv.className = 'message-time full-time';
                            fullTimeDiv.textContent = this.dataset.fullTime;
                            this.appendChild(fullTimeDiv);
                            
                            // 设置定时器，3秒后自动隐藏
                            setTimeout(() => {
                                if (fullTimeDiv.parentNode) {
                                    fullTimeDiv.remove();
                                }
                            }, 3000);
                        }
                    });
                    
                    // 添加到文档片段
                    fragment.appendChild(timeLabelDiv);
                }
                
                // 创建消息容器
                const messageContainer = document.createElement('div');
                messageContainer.className = `message-container ${sender === 'me' ? 'my-message-container' : 'their-message-container'}`;
                
                // 创建消息内容
                const messageDiv = document.createElement('div');
                messageDiv.className = `message ${sender === 'me' ? 'my-message' : 'their-message'}`;
                
                if (type === 'image') {
                    // 图片消息
                    messageDiv.innerHTML = `
                        <div class="message-text">
                            <img src="${content}" alt="图片" class="message-image">
                        </div>
                    `;
                } else if (type === 'emoji') {
                    // 表情消息
                    // 检查是否为图片表情
                    if (content.includes('<img')) {
                        // 如果是图片表情，使用合适的样式
                        messageDiv.innerHTML = `
                            <div class="message-text">${content}</div>
                        `;
                    } else {
                        // 如果是普通文本表情，增加字体大小
                        messageDiv.innerHTML = `
                            <div class="message-text" style="font-size: 2em;">${content}</div>
                        `;
                    }
                } else {
                    // 文本消息
                    messageDiv.innerHTML = `
                        <div class="message-text">${content}</div>
                    `;
                }
                
                // 将消息添加到容器中（不添加头像）
                messageContainer.appendChild(messageDiv);
                
                // 添加到文档片段
                fragment.appendChild(messageContainer);
                
                // 一次性添加所有元素到DOM，减少重排和重绘
                domCache.chatMessages.appendChild(fragment);
                
                // 优化滚动操作：确保新消息始终可见
                requestAnimationFrame(() => {
                    domCache.chatMessages.scrollTop = domCache.chatMessages.scrollHeight;
                });
            });
            
            // 将消息添加到全局数组，避免后续保存时遍历DOM
            messages.push({
                content: content,
                sender: sender,
                time: now.toISOString(),
                type: type
            });
            
            // 更新最后一条消息的时间
            lastMessageTime = currentTime;
            
            // 立即保存到本地存储
            saveToLocalStorage();
        }

        // 发送随机回复
        function sendRandomReply() {
            // 随机选择1-2条回复
            const numReplies = Math.floor(Math.random() * 2) + 1;
            
            // 立即显示正在输入动画
            showTypingIndicator();
            
            let totalDelay = 0;
            let validReplyFound = false;
            
            for (let i = 0; i < numReplies; i++) {
                // 随机选择文本或图片
                const randomType = Math.random();
                
                // 随机间隔时间（1-3秒），使对话更加真实
                const randomDelay = Math.floor(Math.random() * 2000) + 1000;
                totalDelay += randomDelay;
                
                // 总是先尝试文本回复
                if (randomType < 0.8 && phrases.text.length > 0) {
                    // 文本回复（80%概率）
                    const randomIndex = Math.floor(Math.random() * phrases.text.length);
                    setTimeout(() => {
                        hideTypingIndicator();
                        addMessageToChat(phrases.text[randomIndex], 'them', null, 'text');
                        saveToLocalStorage();
                        validReplyFound = true;
                        
                        if (i !== numReplies - 1) {
                            showTypingIndicator();
                        }
                    }, totalDelay);
                } else if (randomType < 0.9 && localImages.length > 0) {
                    // 图片回复（10%概率）
                    const randomIndex = Math.floor(Math.random() * localImages.length);
                    setTimeout(() => {
                        hideTypingIndicator();
                        addMessageToChat(localImages[randomIndex], 'them', null, 'image');
                        saveToLocalStorage();
                        validReplyFound = true;
                        
                        if (i !== numReplies - 1) {
                            showTypingIndicator();
                        }
                    }, totalDelay);
                } else {
                    // 确保即使没有找到合适的回复，也会在延迟后隐藏打字指示器
                    setTimeout(() => {
                        hideTypingIndicator();
                        
                        // 如果还没有找到有效回复且是第一条消息，尝试强制发送文本回复
                        if (!validReplyFound && i === 0 && phrases.text.length > 0) {
                            const randomIndex = Math.floor(Math.random() * phrases.text.length);
                            addMessageToChat(phrases.text[randomIndex], 'them', null, 'text');
                            saveToLocalStorage();
                            validReplyFound = true;
                        }
                        
                        if (i !== numReplies - 1) {
                            showTypingIndicator();
                        }
                    }, totalDelay);
                }
            }
            
            // 最后设置一个安全超时，确保打字指示器一定会被隐藏
            // 这个超时应该比最长可能的回复延迟更长
            const maxPossibleDelay = numReplies * 3000 + 1000;
            setTimeout(() => {
                hideTypingIndicator();
            }, maxPossibleDelay);
        }

        // 发送消息
        function sendMessage() {
            const input = document.getElementById('messageInput');
            const message = input.value.trim();
            
            if (!message) return;
            
            if (continuousSendMode) {
                // 持续发送模式：添加到待发送列表
                addToPendingMessages(message);
            } else {
                // 普通模式：直接发送
                addMessageToChat(message, 'me', null, 'text');
                saveToLocalStorage();
                const delay = parseInt(document.getElementById('replyDelay').value) * 1000;
                setTimeout(() => { sendRandomReply(); }, delay);
            }
            
            // 清空输入框
            input.value = '';
        }

        // 一起发送所有待发送消息
        function sendAllMessages() {
            if (pendingMessages.length === 0) return;
            
            // 发送所有消息
            pendingMessages.forEach(msg => {
                addMessageToChat(msg, 'me', null, 'text');
            });
            
            // 保存到本地存储
            saveToLocalStorage();
            
            // 清空列表并关闭持续发送模式
            clearPendingMessages();
            toggleContinuousSend();
            
            // 获取回复延迟
            const delay = parseInt(document.getElementById('replyDelay').value) * 1000;
            setTimeout(() => { sendRandomReply(); }, delay);
        }

        // 添加本地图片
        function addLocalImage(file) {
            if (!file.type.startsWith('image/')) {
                alert('请选择图片文件');
                return;
            }
            
            const reader = new FileReader();
            reader.onload = function(event) {
                const imageData = event.target.result;
                localImages.push(imageData);
                renderLocalImages();
                saveToLocalStorage();
            };
            reader.readAsDataURL(file);
        }

        // 渲染本地图片
        function renderLocalImages() {
            const imageGrid = document.getElementById('imageGrid');
            if (!imageGrid) return;
            
            // 清空现有图片，保留添加按钮
            imageGrid.innerHTML = '';
            
            // 添加现有图片
            localImages.forEach((imageData, index) => {
                const imageItem = document.createElement('div');
                imageItem.className = 'image-item';
                imageItem.innerHTML = `<img src="${imageData}" alt="本地图片">`;
                
                // 点击发送图片
                imageItem.addEventListener('click', function() {
                    sendImage(imageData);
                });
                
                // 长按删除图片
                let pressTimer;
                imageItem.addEventListener('touchstart', function(e) {
                    pressTimer = setTimeout(function() {
                        if (confirm('确定要删除这张图片吗？')) {
                            localImages.splice(index, 1);
                            renderLocalImages();
                            saveToLocalStorage();
                        }
                    }, 1000);
                    e.preventDefault();
                });
                
                imageItem.addEventListener('touchend', function() {
                    clearTimeout(pressTimer);
                });
                
                imageGrid.appendChild(imageItem);
            });
            
            // 添加添加按钮
            const addImageItem = document.createElement('div');
            addImageItem.className = 'image-item empty';
            addImageItem.id = 'addImageItem';
            addImageItem.innerHTML = '<i class="fas fa-plus"></i>';
            addImageItem.addEventListener('click', function() {
                document.getElementById('imageFileInput').click();
            });
            
            imageGrid.appendChild(addImageItem);
        }

        // 添加新词条
        function addNewPhrase() {
            const promptText = '请输入文本词条：';
            const phrase = prompt(promptText);
            
            if (phrase && phrase.trim() !== '') {
                phrases.text.push(phrase.trim());
                renderPhrases();

        // 编辑词条
        function editPhrase(index) {
            const currentPhrase = phrases.text[index];
            const promptText = '编辑文本词条：';
            const newPhrase = prompt(promptText, currentPhrase);
            
            if (newPhrase && newPhrase.trim() !== '') {
                // 更新数据
                phrases.text[index] = newPhrase.trim();
                
                // 直接更新DOM，而不是重新渲染所有词条
                const phraseItems = document.querySelectorAll('.phrase-item');
                
                // 找到对应的词条（基于data-index属性）
                for (let i = 0; i < phraseItems.length; i++) {
                    const itemIndex = parseInt(phraseItems[i].querySelector('.edit-phrase').getAttribute('data-index'));
                    if (itemIndex === index) {
                        const phraseText = phraseItems[i].querySelector('.phrase-text');
                        if (phraseText) {
                            phraseText.textContent = newPhrase.trim();
                        }
                        break;
                    }
                }
                
                // 保存到localStorage
                saveToLocalStorage();
            }
        }

        // 删除词条
        function deletePhrase(index) {
            if (confirm('确定要删除这个词条吗？')) {
                // 更新数据
                phrases.text.splice(index, 1);
                
                // 直接更新DOM，而不是重新渲染所有词条
                const phraseItems = document.querySelectorAll('.phrase-item');
                
                // 找到对应的词条（基于data-index属性）
                for (let i = 0; i < phraseItems.length; i++) {
                    const itemIndex = parseInt(phraseItems[i].querySelector('.edit-phrase').getAttribute('data-index'));
                    if (itemIndex === index) {
                        phraseItems[i].remove();
                        break;
                    }
                }
                
                // 更新所有后续词条的data-index属性
                const searchTerm = document.getElementById('phraseSearch').value.toLowerCase();
                for (let i = 0; i < phraseItems.length; i++) {
                    const editBtn = phraseItems[i].querySelector('.edit-phrase');
                    const delBtn = phraseItems[i].querySelector('.delete-phrase');
                    if (editBtn) {
                        const itemIndex = parseInt(editBtn.getAttribute('data-index'));
                        if (itemIndex > index) {
                            // 更新data-index属性
                            editBtn.setAttribute('data-index', itemIndex - 1);
                            if (delBtn) {
                                delBtn.setAttribute('data-index', itemIndex - 1);
                            }
                        }
                    }
                }
                
                // 保存到localStorage
                saveToLocalStorage();
            }
        }

        // 过滤词条
        function filterPhrases() {
            const searchTerm = document.getElementById('phraseSearch').value.toLowerCase();
            
            document.querySelectorAll('.phrase-item').forEach(item => {
                const text = item.querySelector('.phrase-text').textContent.toLowerCase();
                if (text.includes(searchTerm)) {
                    item.style.display = 'block';
                } else {
                    item.style.display = 'none';
                }
            });
        }

        // 将十六进制颜色转换为RGB值
        function hexToRgb(hex) {
            // 移除#号
            hex = hex.replace('#', '');
            
            // 解析RGB值
            const r = parseInt(hex.substring(0, 2), 16);
            const g = parseInt(hex.substring(2, 4), 16);
            const b = parseInt(hex.substring(4, 6), 16);
            
            return `${r}, ${g}, ${b}`;
        }

        // 从文本导入数据
        function importDataFromText() {
            try {
                const text = document.getElementById('importTextArea').value;
                if (!text) return;
                
                // 支持所有类型的换行符：\n、\r\n、\r
                const lines = text.split(/\r?\n|\r/).filter(line => line.trim() !== '');
                
                if (lines.length === 0) {
                    alert('没有可导入的有效文本行！');
                    return;
                }
                
                // 将每行文本作为一条消息添加到词条库
                lines.forEach(line => {
                    phrases.text.push(line.trim());
                });
                
                // 保存到本地存储并重新渲染
                saveToLocalStorage();
                renderPhrases();
                
                // 关闭弹窗
                closeImportModal();
                
                // 添加导入成功的用户反馈
                alert(`成功导入 ${lines.length} 条消息！`);
            } catch (error) {
                console.error('导入数据时出错:', error);
                alert('导入数据时出错，请重试！');
            }
        }


        

        


        // 导入数据
        function importData() {
            // 创建一个隐藏的文件输入元素
            const fileInput = document.createElement('input');
            fileInput.type = 'file';
            fileInput.accept = '.json';
            fileInput.style.display = 'none';
            
            try {
                // 添加到文档中
                document.body.appendChild(fileInput);
                
                // 触发点击事件
                fileInput.click();
                
                // 处理文件选择
                fileInput.onchange = function(event) {
                    try {
                        const file = event.target.files[0];
                        if (!file) {
                            // 用户取消选择文件
                            return;
                        }
                        
                        // 验证文件扩展名
                        if (!file.name.endsWith('.json')) {
                            alert('导入失败：请选择JSON格式的文件。');
                            return;
                        }
                        
                        const reader = new FileReader();
                        reader.onload = function(e) {
                            try {
                                const importedData = JSON.parse(e.target.result);
                                let importedItems = [];
                                
                                // 处理导入的数据
                                if (importedData.messages && Array.isArray(importedData.messages)) {
                                    localStorage.setItem('Q_messages', JSON.stringify(importedData.messages));
                                    importedItems.push('聊天记录');
                                    
                                    // 清空当前聊天界面
                                    const chatMessages = document.getElementById('chatMessages');
                                    chatMessages.innerHTML = '';
                                    
                                    // 清空全局messages数组
                                    messages = [];
                                    
                                    // 重新渲染所有导入的消息
                                    importedData.messages.forEach(msg => {
                                        // 兼容旧的text属性和新的content属性
                                        const content = msg.content || msg.text || '';
                                        addMessageToChat(content, msg.sender, msg.time, msg.type || 'text');
                                    });
                                }
                                
                                if (importedData.phrases) {
                                    phrases = importedData.phrases;
                                    // 保存phrases到localStorage
                                    localStorage.setItem('Q_phrases', JSON.stringify(phrases));
                                    renderPhrases();
                                    importedItems.push('词条库');
                                }
                                
                                if (importedData.localImages && Array.isArray(importedData.localImages)) {
                                    localImages = importedData.localImages;
                                    // 保存localImages到localStorage
                                    localStorage.setItem('Q_localImages', JSON.stringify(localImages));
                                    renderLocalImages();
                                    importedItems.push('本地图片');
                                }
                                
                                if (importedData.settings) {
                                    localStorage.setItem('Q_settings', JSON.stringify(importedData.settings));
                                    importedItems.push('设置');
                                }
                                
                                // 处理设置的应用
                                if (importedData.settings) {
                                    // 重新加载设置
                                    loadFromLocalStorage();
                                }
                                
                                // 保存所有数据变更
                                saveToLocalStorage();
                                
                                alert(importedItems.length > 0 ? `导入成功！已导入：${importedItems.join('、')}` : '导入成功！但未发现可导入的数据。');
                            } catch (error) {
                                console.error('JSON解析错误：', error);
                                alert('导入失败：JSON格式错误，请检查文件内容。');
                            }
                        };
                        reader.readAsText(file);
                    } catch (error) {
                        console.error('文件处理错误：', error);
                        alert('导入失败：文件处理错误，请重试。');
                    } finally {
                        // 确保文件输入元素被移除
                        if (fileInput.parentNode) {
                            document.body.removeChild(fileInput);
                        }
                    }
                };
            } catch (error) {
                console.error('创建文件输入错误：', error);
                alert('导入失败：无法创建文件选择器，请重试。');
                // 确保文件输入元素被移除
                if (fileInput.parentNode) {
                    document.body.removeChild(fileInput);
                }
            }
        }
